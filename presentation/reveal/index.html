Inheritance
class Color {
	color = "white";
}

class Shape extends Color {
	constructor(shape) {
		super();
		this.shape = shape;
	}
}

const circle = new Shape("circle");
circle.color; // white

const square = new Shape("square");
square.color; // white

const triangle = new Shape("triangle");
triangle.color; // white

Composition
class Color {
	color = "white";
}

class Shape {
	colorInstance = new Color();
	constructor(shape) {
		this.shape = shape;
	}
}

const circle = new Shape("circle");
circle.colorInstance.color; // white

const square = new Shape("square");
square.colorInstance.color; // white

const triangle = new Shape("triangle");

class Color {

	log() {
		return "from color";
	}
}

class Shape extends Color {
	id = "shapeId";
	constructor(shape) {
		super();
		this.shape = shape;
	}

	log() {
		return "from shape";
	}
}

const circle = new Shape("circle");
circle.id; // shapeId
circle.log(); // from shape

class Color {
	id = "colorId";
	color = "white";

	log() {
		return "from color";
	}
}

class Shape {
	colorInstance = new Color();
	id = "shapeId";
	constructor(shape) {
		this.shape = shape;
	}

	log() {
		return "from shape";
	}
}

const circle = new Shape("circle");
circle.colorInstance.color; // white

circle.colorInstance.id; // colorId
circle.id; // shapeId

circle.colorInstance.log(); // from shape
circle.log(); // from color
